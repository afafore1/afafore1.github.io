Simple Calculation

! y = x^3 - 14x^2 + 56x - 64
define(x_r, l0)
define(y_r, l1)
define(a0,14)
define(a1,56)
define(a2,64)

.global main
main:
        save %sp, -96, %sp

        mov 3, %x_r
        mov %x_r, %o0
        call .mul
        mov %x_r, %o1
        call .mul
        mov %x_r, %o1
        mov %o0, %y_r !mov x^3 into %y_r

clr %x_r
        mov 3, %x_r
        mov %x_r, %o0
        call .mul
        mov %x_r, %o1
        call .mul
        mov a0, %o1 ! 14x^2
        sub %y_r, %o0, %y_r ! x^3-14x^2 and move result in %y_r

        clr %x_r
        mov 3, %x_r
        mov %x_r, %o0
        call .mul
        mov a1, %o1 !56x
        add %y_r, %o0, %y_r ! x^3-14x^2+56x and move result in %y_r

        sub %y_r, a2, %y_r ! x^3-14x^2+56x-64 and answer in %y_r

        mov 1, %g1
        ta 0

Max Find

arr_offset = -5*4
max_offset = arr_offset - 4
size_int = 4

.global main
main:
        save %sp, (-92+max_offset)&-8,%sp
        mov 2, %o0
        st %o0, [%fp + arr_offset]
        mov 1, %o0
        st %o0, [%fp + arr_offset + 1*size_int]
        mov 8, %o0
        st %o0, [%fp + arr_offset + 2*size_int]
        mov 16, %o0
        st %o0, [%fp + arr_offset + 3*size_int]
        mov 21, %o0
        st %o0, [%fp + arr_offset + 4*size_int]

        ld [%fp + arr_offset], %l0

        mov 4, %l1
        mov 0, %l2
        ba looptest
        cmp %l1, %l2

        forloop:
        ld [%fp + %l2], %o0
        cmp %o0, %l0
        bl looptest
        deccc %l1 !index = index + 1
        mov %o0, %l0

        looptest:
        bge, a forloop
        dec size_int, %l2

        st %l0, [%fp + max_offset]
        mov 1, %g1
        ta 0

Swap Subroutine

.global swap
swap:
save %sp, -96, %sp
mov %i0, %l0
mov %i1, %i0
mov %l0, %i1
ret
restore

.global main
main:
save %sp, -96, %sp
mov 5, %o0
mov 7, %o1
call swap
nop
mov 1, %g1
ta 0
