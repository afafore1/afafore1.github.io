Sieve of Erathoneses

#include 
#include 

#define limit 2000000

int main()
{
        unsigned long long int i, j, answer; //declare variables i and j this way because they are going to be huge
        int primes[limit] = {}; //store numbers in array
        answer = 0; // initialize answer

        for (i = 1; i < limit; i++){ // start from 2 since every number is a multiple of 1.
        primes[i] = 1; } //store first number as prime. In our case 2
        for (i = 2; i <= limit; i++){
                if (primes[i-1]){// i is prime if it's not crossed out yet
                        answer = i + answer; // answer is equal sum of all primes
                        for (j =i*i ; j <= limit; j+=i){// cross out all multiples of primes
                        primes[j-1] = 0;
                                        }
                                }
                }

printf("%ld\n", (long) answer); // print result
        return 0;
}
